```php
<?php
$INFO = [
    'host'      => '',//host mail server
    'username'  => '',
    'password'  => '',
    'port'      => '587',
    'type'      => 'tls',//tls SMTPSecure
    'to_mail'   => '', // chi riceve la mail dal contact.php
    'to_name'   => '' //nome che compare nel mittente
];
```
